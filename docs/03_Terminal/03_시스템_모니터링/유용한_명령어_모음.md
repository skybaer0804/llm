## 유용한 명령어 모음

```bash
# Ollama 관리
ollama list                                    # 설치된 모델 확인
ollama pull <model-name>                       # 모델 다운로드
ollama rm <model-name>                         # 모델 삭제
ollama run <model-name>                        # 모델 실행
brew services list                             # 서비스 상태 확인
brew services restart ollama                   # Ollama 재시작

# Docker 관리
docker ps                                      # 실행 중인 컨테이너
docker logs -f <container-name>                # 실시간 로그
docker restart <container-name>                # 컨테이너 재시작
docker system prune                            # 정리

# Git 관리
gh issue list --state=open                     # 열린 이슈 확인
gh issue view <issue-number>                   # 이슈 상세 보기
gh issue comment <issue-number> -b "메시지"    # 댓글 추가

# 모니터링
top                                            # CPU/메모리
df -h                                          # 디스크
log stream --predicate 'process == "ollama"'   # Ollama 로그
tail -f /tmp/autogen.log                       # AutoGen 로그

# FastAPI 라우터 실행
python -m uvicorn main:app --reload --host 127.0.0.1 --port 8000
```
